create table public.case_documents (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  case_id uuid null,
  file_path text null,
  file_name text null,
  file_type text null,
  file_size bigint null,
  document_type text null,
  uploaded_by uuid null default gen_random_uuid (),
  uploaded_at timestamp with time zone null,
  metadata jsonb null,
  constraint case_documents_pkey primary key (id),
  constraint case_documents_case_id_fkey foreign KEY (case_id) references cases (id) on update CASCADE on delete CASCADE,
  constraint case_documents_uploaded_by_fkey foreign KEY (uploaded_by) references user_profile (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.cases (
  id uuid not null default gen_random_uuid (),
  user_id uuid not null,
  title text null,
  description text null,
  status text null,
  metadata jsonb null,
  created_at timestamp with time zone null default now(),
  updated_at timestamp with time zone null default now(),
  boldsign_document_id text null,
  signature_status text null default 'pending'::text,
  signature_completed_at timestamp with time zone null,
  call_details jsonb null,
  call_summary jsonb null,
  document_summaries jsonb null,
  followups jsonb null,
  constraint cases_pkey primary key (id),
  constraint cases_user_id_fkey foreign KEY (user_id) references user_profile (user_id) on delete CASCADE
) TABLESPACE pg_default;

create table public.notifications (
  id uuid not null default gen_random_uuid (),
  created_at timestamp with time zone not null default now(),
  user_id uuid null,
  title text null,
  type text null,
  message text null,
  data jsonb null,
  read boolean null,
  constraint notifications_pkey primary key (id),
  constraint notifications_user_id_fkey foreign KEY (user_id) references user_profile (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.user_eligibility (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  uploaded_file text null,
  eligibility_rating integer null,
  eligibility_title text null,
  eligibility_message text null,
  eligibility_confidence integer null,
  eligibility_raw jsonb null,
  processed_at timestamp with time zone null default now(),
  case_id uuid null,
  constraint user_eligibility_pkey primary key (id),
  constraint user_eligibility_case_id_fkey foreign KEY (case_id) references cases (id) on update CASCADE on delete CASCADE,
  constraint user_eligibility_user_id_fkey foreign KEY (user_id) references user_profile (user_id) on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_user_eligibility_user_id on public.user_eligibility using btree (user_id) TABLESPACE pg_default;

create table public.user_profile (
  id uuid not null default gen_random_uuid (),
  user_id uuid null,
  full_name text null,
  email text null,
  phone text null,
  identity_code text null,
  email_otp text null,
  otp_expires_at timestamp with time zone null,
  verified boolean null default false,
  eligibility_rating integer null,
  eligibility_title text null,
  eligibility_message text null,
  eligibility_confidence integer null,
  eligibility_raw jsonb null,
  created_at timestamp with time zone null default now(),
  is_admin boolean null,
  role text null,
  is_subadmin boolean null,
  photo_url text null,
  payments jsonb null,
  contact_details jsonb null,
  constraint user_profile_pkey primary key (id),
  constraint user_profile_email_key unique (email),
  constraint user_profile_user_id_key unique (user_id),
  constraint user_profile_user_id_fkey foreign KEY (user_id) references auth.users (id) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create index IF not exists idx_user_profile_email on public.user_profile using btree (email) TABLESPACE pg_default;
